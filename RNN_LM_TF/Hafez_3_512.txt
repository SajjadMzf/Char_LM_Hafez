
ما را نسيم مباد حافظ	
	  	
گشت و ره قدح گر برآن کنم گويند	اي پادشاه دار من افتد که جمشيد	
که دوشتن در کنج سرگردون سوي سوي تو مدارم	من از ديدة ما جام ليکن نگون داشند	
کان شوخ سيغ بيني در من مشکين غم مخور	اي دل ار خمار اندر طلبت چو نيست	
گرچه افشاند در سر مي کنند مي رود	اش حافظ چو طالب تو اي سين سرگشته ما نيست	
	  	
دوستد و تنگيم و نام و ناله مي روم	خرقة چشم مشتان نهاده ايم تو	
مي خواهد که مست و نياز خراب کن	با خضر نشين با من درويش نگفت	
گل خرم آن سراک و همي نگردد مبارا	مکا را نياز مي گويم چه صوفي را شد کنم	
تا گوي ز دست برآري که چه داني چه نوش	من از دست چون شمع از دست برشد	
که حافظ را به بلبل دارگ جلال	هرگز نز سر فقر به دست صاحب 	

	  	
رخ صومعه اي ز هاده از جائي چه خانه	شراب الف سلطان را چه حاجت است 	
زان سر ز ره و سسر شراب جامي	چون مرا مي خواند معاشي ريخت خون ما 	
هر دهد ديده از ني ز ني ني داري	دل چو آشيان نرگس جادا را 	
به جائي رسيد مدابان نيست	در ساي ما همچو جام مي از دست 	
کزان که نگاري کاسر سر ز روي خود 	اي دل چه حاجت است در سر جاني مي 	
هاي خوش آن ناز هزار گشته با انداشي	دل مزنجائي چو حافظ مرغين در ايم	
	  	
مست رمي گو بر سر ناز پر نيست	دلم ز دو قند پير مغان شيني	
مي دهد دانة او خوش با مي راي	زوهد العمات چو چند برخوز با همچو 	
که دگر نيست ره به مي اندش	بي ره ميخانه در مستي نهگاني	
مي دهد دارد همي گويم شد مدام ما	از عاشقان نبرد و لطيف مات مستي	
زان شماي دلبر نديده چو تو ار شوي	در اين نعته که در ميان ساغي مست	
که دوست داد آن شهد که نمي خوشي	در خانة اشکار اسم انعام را 	
در ديار است در ساينة تيريست	دل شيخوري اندام ما در سينه نام ما	
من انديشه خاني ار سيم اندازي	معنق از دست گو بدان دام را	
خوش آمان شاه شي ساغر شدي	اي دل با دلدار ديگر من از جاني	
که دردند و تو در شمع نام را	در عيش نه نام و قصد جز نام ما	
چرخ ارم به نور نمي دهي ديد پايد	به به نوش ز صد بار هشچاه و خواب	
در ساية اسرار دوران نيست در انداري	دل به جان را نسيم ما نامان شدم	
تا خون شاي سروش کار جهان را چه ني	ترا که دست از خون دل مستين نشاني	
که دل به نام نيست رخ هرچه  اي ديني	نه هر که در ما مي ده ترسم زان که 	
مرا نيست نديدي و در سايان داري	تيه لعل و آن عارض قدم را ندارد 	
همچو توئي گر نمي روي ز آه خدني	دل به دست انديشه که در مستي	
که در مشاطا عشق دامن چاک را	ز من ز رقي است در دست نام کنيد	
که نه بر سر ديده است اگر در آن تو نامارا	مگر دينانه ها دست اهن نقش نيست 	
هم ز دور نفاي خير ي دهرش نام 	در دست را ز نه و روي حاقت را	
چو بر فلک نشکند الحکن شکاري	دل مي رود از دست را دوام را 	
که دور زمانه فسنگ و گني در ما	عيان اين شد و گريه نقش مي آرد	
که مرغ نقد رو دردست داداري	ز راز به شود ساقي ره ندانست ند	
که همچو چشم تو در جيانان شما را 	ما در چشم و مي است رندي به نيم 	
عاقلان سرو خداوار ساکي	دوش گر مه بر نود مي داري	
وگر و گلين تو خوشند شاهي نام ما	اي سلطان مگر گرده ها به را 	
سرم که چو بنگاهي به نام ما	اي ساقي به دست از ننگشت دريا 	
هده شاه درين ره نه نهاني	مرا در عيش نورديد و معتور نيست	
از چاه نمي کني اي قيري نيست 	 راحد در سينه داران مي داري	
به خون شد قلم قيام و نهاري نمي نام ما	دل در سينه هوسي نه بلباني	
که با درد من دل ساية عشقت چو مه ويد 	رندي ز عشق و دل با دل دلداري 	
عاقلا ما ز سراي سرا را 	مرا در عاشقان قدس زان دردن فروز	
که در شعاج تو روزي کرم و از نام ما	در دوم عاقبت ما زد به شادي	
که بر سر اين فنائي نه در استانه 	اين همه دست نامي نيست نام کن	
که نمي خواهي از باد دهنائي داني	به عشق من سري از خدا شود در سوني	
نکم که در منت است در ساية عشق نشودي	دل من در دست و داما و گريان اي دارد 	
که بنگشتي نه دل دردان دارانند	جاي دهاني نيک تو در خاک سر ما را	
مرا در نام من نام کن اگر ني داري	دل به جان را نسيم ما نامه سينه ندهم	
ره نهانست نه در در ميکده نام ما	در دوم رند شود هر که چه مي گوئي	
کز ناز بختم که دم از دست دارائي	چو گنبد از به چنگ ار دلي نمي گيري	
ورنه در دست او مي رسي اي صاحب جمائي	در زمانه سوداي دوست نقش مستي را 	
که بس نيست نهان آستينه ناز مست	اگر ند دامن چو ماهد آن شکند 	
که صد مي شند ندارد از خون است ناي داني	به دست دام جهد من است در پنهان 	
که نديديم دل باده نگاني داري	دلي که داني ناز يا دهند پاياني	
و نه با ديده با در وقتي نيست	من نه بنگر برنشدند از عشق نگران	
اي چنين چه سربلند ديد را نهاديم	زلف تو در خاتم در دست داده ريخت	
که در ميان تو با ما نيست دامان کرم	رونق چنم از خاک در مستي نيست	
چه نامه کند داين در سينه ندهم	مرا در قيام و دور نماندي رخت داني	
کز ناه دلندان در دست نازاهي	در دم از ما در طلب کند نامه کنم ره دانا	
چاندينه در ناتواني داني دارد	صبا حافظ و آهش جام جا از خانقان	
	  	
دهان نيم شد از دستت را ياران 	مي دهد خدايا رها دوست ناشود	
که داشت در سر اشکاري دانا باشد	دل به مرغ ناز و معشوق منظر دارا	
که نيست در خون دل آن سرا و خون داني	نيست نگارا و نازد دل در کس نيک	
که نديديد و معنايي عيار شاد	بگذر دل شد از لطف چنگ دارا	
ز ما در منش نامان نظر نيست	ما در دم از ما نيست در است نايم	
که نيستشد را ز نظر باده داري	دون داد و دل نازنيني نيک دانا	
ها در خرمن نامه سندن دردي	نقد دوست ندارد قصد داما را	
خوشاندي که داني آنجا نيست نيم را	اگر نه با دل بي طرب چو مرغان 	
که با دلم ما هر سداياني داني	ز نه رقيب نما در سينه نيست	
چنگ مهر و ما نيست در سر ايام ما	اي که با دلدار ياد خاني و نه عجب 	
زانها به لب لب لعل تو حق نکنيم	ما نگوئيد خواب و مستي را همچو حاجي	
که داني ندارد ز بند باده نيست	از ديان نازه نهانيست نمي دهد	
روزي را که نمي توان داري دارد	چه گويم که در سياه مي گفت دانا	
ز ني در خون دل نشسته هزار ناله مکن	چنين چه دانش که همدار دگر نه نه	
که نداني که به جان را چه ناشاني	در دست رند و به ره دوست نداري داري	
که داند ره نيست نمي بايد داري	ز من به دست حافظ دل در دوران	
	  	
خوش باش از عاشقان نيست به هر دو داني	به هر کس نگاري بوده اي داند که بوي	
مردم در آن دست که دوران ندادي	سلامت چو گل از دست برشد دانات دارد	
کز من تيغي دل از خونبار دارد	دلي اي دل که چو ما در سر سيلان داري	
که بود در خت به شهود داوادگان شاه	با دل اي من خاکي و نه هر سو داني	
که بدهند و جام نيست نهاني داني	چو خسمور نو به سلامت همه داراني	
که با در وينه در سر افشان باشد	خواجه دوستده رندان که نداني زين دل	
که صد هزار حافظ و اين الحان نيست	ز هنچه سرگشتگان در دلم دارد	
	  	
خاک نازه نيازين را خوبي داري	خد ما که بندي از خون دل پر شد 	
ما نگردد همچو من ديده بر حسن را دريا دارا	همه رخ و خوبان مي دارد راه ساکراني	
که به جوي نوستي از دست باشد چو من است	در دل نازک نظرم در دهان 	
جهان بود از هجران که نه دل اي حافظ	زمانه دردانه داري چه داني که ناز	
	  	
خاك را نه سر از باد صبا که نيست	اي کن نوان شکند رندان رسيد ازي	
که ناز غم در ساية دلهاي تو رو	گر دولت از خون مي داني که نه نيم شي	
عيش نيست نگوئي که چه در و راه خواهد کرد	عيب نظري بخشد از لطف ندادم چو عنق اي	
که به ما نيست که در دست ندارد هيچي	ساني که دردان داني نازين است نگاراني	
که نيست در سر او وانجه تا باره جاناني	خدا را در سرکشان بردند حافظ	
	  	
خوبش باد که نوسته اند و ده نه يک اي خواني	هر که در دل به قول كه دست دادا و ني	
در سوي عيش نمي شود آه شب ماهي	دلم ز من به خون ديشي و مطرب نيست	
به نام من در سر سر فروان چين 	مستي از دل نازه نازان شنيد ماناني	
تو از اما پيش مي گذاري بنواده دستي	خون در پاي تو در دست و ساي باشي ناشيني	
در نظر آشنائش چند باشد چو مي دهد	چشم دراز دست و داده رشک روي رسيد	
کز ساغر ما برگ نمي آند رقم را 	 اي شنيدي در عشق شيخ ساغري	
که گفت در شيخ مي نمي توسي غنيم ري	ز دست برش رمان بر دانا داري	
مگر از مي گويم خون نه مي نيست که مي	پرديم و مهر و مطرب نمي داني مي راه مي داري	
اگر خود را به چنگ ار بندي روزي نيست	دل بر من گر ز نور دهين پر ري	
گر چه نيم شد از مي نوش خواهي مي مي ديديم	ز باده راه دهان نقشت دردانج ستنده	
که در نيار در خواني نوش نديديم و خواب	زده خاک ور دوگردان چو بنديده برون شيد	
که بر ساغن با مستي به جست آي شد ماهر مي	حافظ چنان به که دوست در سر جهان به دوست	
	  	
گر نشيند شود در سهن ساني به ما نمي نيزد	مرا چون برقي از دل مي شود نه سر بيار	
ناز مي دهند از نظم رازي نيست	دولت روزي چو من آن دل که نديدم را	
دلم از مشاط مصد مستين نشان مي داري	سانها گر با تو گر چه با در کم راه ناماني	
جز من و در مشاطه رخسان کردارانم	چو شمع زنده رود از حافظ سپنه در دل دارا	
	  	
دل ز اندام تو نسيم دامي داري	خون دامي نه نازح نگان داري نيست	
گر سر فروز ساية دانا نامه خواهد بود	عشق از آن ساه تو چون شمع در سينه ندادند	
که نيست رحم و عذار فرماند بر عاشق	دلا را دو عارض بي سر بوي تو باشد	
وز مي زده رو که در استماري خوش	اي دل ديگه گو که به گنج دوست نيست	
در سر اي دل بر سر اين نماند نهاديم	سر شود از زهد مي دهند حافظ مي داري	
	  	
برگ نه شنود هر که پشيماني نرگس منان مي داري	زين قده را نيارد در دل برون نمي گوي	
ز مهر او نه ديده را نور نيست چمي	چو ناصه بر دل شادي اي دل بي نظاري	
که درد عشقبازي اماني نه سر بپوشاني	چو مهر لب مي فرماند هر سمنش ناز	
که بر نام جانا نه در اياني دارد	شب روزي بساز از دل بي نجازي برخانم	
من که درداني نشود آهي رخت داني	در چمن هر سو که در ميخانه خيالش	
تا خود در سر اين معما نه برگاهي	ناه ندانست که دل از شعر اي خواهي در دم	
هزار جان فراوي نخواهد ماند	ما در دست رندي و مهر و خون دل پاي	
چه بوي من ندهد کند قدم ره دل داراي	ز من به خون دل بشوئي من سرگشته 	
که ناز نياست به ابر راهي نيست	در چمن هر سويد باش تنگارساک	
که در دل مي گزي که چون ساغر نمي ارزد	چو منزويست در خرقه آن ساکه پرساني شمن	
که نيست در خندة صوفي به دوم نيست	در چمن ناهد کانجا ز ناز دمينا چمن	
که در گوش مي دست ناگه خواهد مي مي داري	ز من به دست من از ما نيست در پنه عديد	
هر که چون او داني زين خراب ما ره نکند	ما در ره از دل من مرغ نيست نداري	
که نيست در خون دل پيش نيست در دل دارا	ز من اگر نه با هوا بود هم سبنگاني	
که در محامي در تمنا که دهداني	دون مه به ناز است روزي بر نقد عيش دادائي	
ورنه در دوست ندارد پس هائي دارد	دلم که صافي در ميخانه بيمار ندارد	
که در نظر آن سر از ساحر مي داني نيم	در دل ديده به جوي خود نهاني	
مکار جمعه است در آن دانه ندارد	جدا را کم نشين با در انديشه نقاب	
که در آن تست که دگر نشيد باد هاچي	چو حافظ نبود که دردانه دهد درجان داري	
	  	
هر زمانه به مي گذرد پير منداري مي روي	چو انديشه نقشي دارد از دستم صاحبدلاني 	
که جوش بي سراوي ندهد آشکاري خوش	شبي از دست ناگود و گنج طرياب 	
به سيو باشد چه دومي گري دهاني	يا ره که نيست ندارم چو گذري بر نه	
نگان به شراب اندر نظر نيست نالا نيست زانکه نگشتيم	جويب را دوست ناميت تن در رضي 	
که صبح مي دهد دل در خواب و ديده بنشان	شد و گوي عيش کند از رنگ و راه ماهي	
ورنه هر ساني که دهد از نيازي داري	چو گل از بهاي منت ار در خواهي نيست داري	
تا خواب مي دهد در سر ديوري بود	ملالي چه دوئي کندي چه گويم که چه باکي	
خلوت گمراه که چون تست پاي داني	دردم از دلبند ما گفته به چنگ است	
وز ناز بر حضور عاقله را در نمي دادي	چشم است و راه قدح را بسي پيش بازي	
که دم زنر ساغرگه پر شاهم از دوست	ني بر عالم ز به درجان نيست	
کاين جلوه ناز رخ رنگين تو نيز داري	کين به من آن دو چشمم ندهد سلولي	
نه در اندانة اسرار دلياه مدامي	چو مرغ در آن حمال ملامت نه ديدناني	
که ناز نخار اي نسيم ابروي رخشيني	مدعون گو چه مي جوئي و مي از راه عشق ناني	
رخ نمي داند رخ ساقي نيست نامي کنند	يا رب چو دل گردم اندر کس نيست	
بر اي قير مرا دل که دهد دامي	ناز نصد حافظ دارد دلي در دهم	

	  	
به نيم شب مي نوش نيز نيم ما	صوفي به ديده بر نازه مي گويم	
ولي شهر و که نمي روي ز پاي را سوزي مي دادي	دلم که بي مرا آن ساهر نه در درگو دارا	
که بنوش مي صوف کسم و نام مي داري	حافظ از مشوق دلدار تو داري دين 	
	  	
در خون شيه و گنه قيري چنين که نيست روي تو خوشياني	شيراز و خالي مهربان که به خون عيب 	
که ندهد عيش نيست در دل خوشيد امت	در کس نشاند ناز و خرد نمي گويم	
به شکر کن و با خود خرامي نيست	اي گونم آن دل که در عشوب شدم	
که در خام رشت و خراب مي دهد پيري	در دل که بنما تو در آن عشق دارد	
سوندار باد عهد کانار دهاني چيزم	در سر ابروي جانان پسر آمده اي يعني چه	
در اي ناز و در خانة عشق نيستي داني	ناز بر صبر و دل در صفا تنه کجاست	
چه بر در ديدة گل نيست چه داني	مگر دور نمي اندر شعر دستم دادي
